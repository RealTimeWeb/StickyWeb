001    package realtimeweb.stickyweb;
002    
003    import java.io.IOException;
004    import java.net.URISyntaxException;
005    import java.util.ArrayList;
006    import java.util.Map;
007    
008    import realtimeweb.stickyweb.exceptions.StickyWebDataSourceNotFoundException;
009    import realtimeweb.stickyweb.exceptions.StickyWebInternetException;
010    import realtimeweb.stickyweb.exceptions.StickyWebNotInCacheException;
011    
012    
013    /**
014     * Class for building up a Web Request; uses the fluent pattern.
015     * @author acbart
016     *
017     */
018    public class StickyWebRequest {
019            
020            private boolean online;
021            private LocalCache cache;
022            private Protocol protocol;
023            private String url;
024            private Map<String, String> arguments;
025            private ArrayList<String> indices;
026            
027            /**
028             * Set whether results should be returned from the web or from the local cache.
029             * @param online
030             * @return
031             * @throws StickyWebDataSourceNotFoundException 
032             */
033            public StickyWebRequest setOnline(boolean online) throws StickyWebDataSourceNotFoundException {
034                    if (!online && cache == null) {
035                            throw new StickyWebDataSourceNotFoundException("The given InputStream was null; check to make sure that the file exists.");
036                    } else {
037                            this.online = online;
038                    }
039                    return this;
040            }
041            
042            /**
043             * Specifies what parameters should be used when hashing a request; that is,
044             * what indexes uniquely identify a call. If null, then all parameters will be used.
045             * @param indexes
046             * @return
047             */
048            public StickyWebRequest setIndexes(ArrayList<String> indexes) {
049                    this.indices = indexes;
050                    return this;
051            }
052    
053            /**
054             * Creates a new StickyWebRequest with parameters.
055             * @param url The url of the resource.
056             * @param arguments Un-encoded data that will be sent along with the request.
057             * @param cache A reference to the local system's cache
058             * @param protocol The method that the resource will be accessed.
059             */
060            public StickyWebRequest(String url, Map<String, String> arguments,
061                            LocalCache cache, Protocol protocol) {
062                    this.url = url;
063                    this.arguments = arguments;
064                    this.cache = cache;
065                    this.protocol = protocol;
066                    this.indices = null;
067            }
068            
069            /**
070             * Gets the response for this request, from either the online service or the local cache.
071             * @return
072             * @throws IllegalStateException
073             * @throws IOException
074             * @throws URISyntaxException
075             * @throws StickyWebNotInCacheException
076             * @throws StickyWebInternetException 
077             */
078            public synchronized StickyWebResponse execute() throws IllegalStateException, IOException, URISyntaxException, StickyWebNotInCacheException, StickyWebInternetException {
079                    if (online) {
080                            switch (protocol) {
081                            default: case GET: return new StickyWebResponse(WebConnection.get(url, arguments));
082                            case POST: return new StickyWebResponse(WebConnection.post(url, arguments));
083                            case PUT: return new StickyWebResponse(WebConnection.put(url, arguments));
084                            case DELETE: return new StickyWebResponse(WebConnection.delete(url, arguments));
085                            }
086                    } else {
087                            return new StickyWebResponse(cache.get(url, arguments, this.indices));
088                    }
089            }
090    
091    }
