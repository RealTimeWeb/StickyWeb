001    package realtimeweb.stickyweb;
002    
003    import java.util.ArrayList;
004    import java.util.Map;
005    
006    import org.scribe.model.Token;
007    import org.scribe.oauth.OAuthService;
008    
009    import realtimeweb.stickyweb.exceptions.StickyWebDataSourceNotFoundException;
010    import realtimeweb.stickyweb.exceptions.StickyWebInternetException;
011    import realtimeweb.stickyweb.exceptions.StickyWebInvalidPostArguments;
012    import realtimeweb.stickyweb.exceptions.StickyWebInvalidQueryString;
013    import realtimeweb.stickyweb.exceptions.StickyWebNotInCacheException;
014    
015    /**
016     * Class for building up a Web Request; uses the fluent pattern.
017     * 
018     * @author acbart
019     * 
020     */
021    public class StickyWebRequest {
022    
023            private boolean online;
024            private LocalCache cache;
025            private Protocol protocol;
026            private String url;
027            private Map<String, String> arguments;
028            private ArrayList<String> indices;
029            private OAuthService service;
030            private Token accessToken;
031    
032            /**
033             * Set whether results should be returned from the web or from the local
034             * cache.
035             * 
036             * @param online
037             * @return
038             * @throws StickyWebDataSourceNotFoundException
039             */
040            public StickyWebRequest setOnline(boolean online)
041                            throws StickyWebDataSourceNotFoundException {
042                    if (!online && cache == null) {
043                            throw new StickyWebDataSourceNotFoundException(
044                                            "The given InputStream was null; check to make sure that the file exists.");
045                    } else {
046                            this.online = online;
047                    }
048                    return this;
049            }
050    
051            /**
052             * Specifies what parameters should be used when hashing a request; that is,
053             * what indexes uniquely identify a call. If null, then all parameters will
054             * be used.
055             * 
056             * @param indexes
057             * @return
058             */
059            public StickyWebRequest setIndexes(ArrayList<String> indexes) {
060                    this.indices = indexes;
061                    return this;
062            }
063    
064            /**
065             * Creates a new StickyWebRequest with parameters.
066             * 
067             * @param url
068             *            The url of the resource.
069             * @param arguments
070             *            Un-encoded data that will be sent along with the request.
071             * @param cache
072             *            A reference to the local system's cache
073             * @param protocol
074             *            The method that the resource will be accessed.
075             * @param accessToken
076             * @param service
077             */
078            public StickyWebRequest(String url, Map<String, String> arguments,
079                            LocalCache cache, Protocol protocol, OAuthService service,
080                            Token accessToken) {
081                    this.url = url;
082                    this.arguments = arguments;
083                    this.cache = cache;
084                    this.protocol = protocol;
085                    this.indices = null;
086                    this.service = service;
087                    this.accessToken = accessToken;
088            }
089    
090            /**
091             * Gets the response for this request, from either the online service or the
092             * local cache.
093             * 
094             * @return
095             * @throws StickyWebNotInCacheException
096             * @throws StickyWebInternetException
097             * @throws StickyWebInvalidQueryString
098             * @throws StickyWebInvalidPostArguments
099             */
100            public synchronized StickyWebResponse execute()
101                            throws StickyWebNotInCacheException, StickyWebInternetException,
102                            StickyWebInvalidQueryString, StickyWebInvalidPostArguments {
103                    if (online) {
104                            switch (protocol) {
105                            default:
106                            case GET:
107                                    return new StickyWebResponse(WebConnection.get(url, arguments,
108                                                    service, accessToken));
109                            case POST:
110                                    return new StickyWebResponse(WebConnection.post(url, arguments,
111                                                    service, accessToken));
112                            case PUT:
113                                    return new StickyWebResponse(WebConnection.put(url, arguments,
114                                                    service, accessToken));
115                            case DELETE:
116                                    return new StickyWebResponse(WebConnection.delete(url,
117                                                    arguments, service, accessToken));
118                            }
119                    } else {
120                            return new StickyWebResponse(
121                                            cache.get(url, arguments, this.indices));
122                    }
123            }
124    
125            /**
126             * Returns the internal hash of the url and its arguments that will be used
127             * to look up the data in the cache.
128             * 
129             * @return
130             */
131            public String getHashedRequest() {
132                    return LocalCache.hashRequest(url, arguments, indices);
133            }
134    
135    }
