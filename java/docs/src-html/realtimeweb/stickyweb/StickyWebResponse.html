001    package realtimeweb.stickyweb;
002    
003    import java.io.IOException;
004    import java.io.StringReader;
005    import java.util.List;
006    import java.util.Map;
007    
008    import org.htmlcleaner.TagNode;
009    import org.json.simple.parser.ParseException;
010    import org.w3c.dom.Document;
011    import org.xml.sax.SAXException;
012    
013    import realtimeweb.stickyweb.converters.HTMLConverter;
014    import realtimeweb.stickyweb.converters.JsonConverter;
015    import realtimeweb.stickyweb.converters.XMLConverter;
016    import realtimeweb.stickyweb.exceptions.StickyWebJsonResponseParseException;
017    
018    import au.com.bytecode.opencsv.CSVReader;
019    
020    public class StickyWebResponse {
021            
022            private String text;
023    
024            /**
025             * Create a new StickyWebReponse from raw data.
026             * @param text
027             */
028            public StickyWebResponse(String text) {
029                    this.text = text;
030            }
031            
032            /**
033             * Occasionally, data will come in malformed for one reason or another.
034             * This function, in conjunction with asText, allows you to manipulate the data
035             * before it is returned through asJSON or another method.
036             * @param newText
037             */
038            public void setText(String newText) {
039                    this.text = newText;
040            }
041    
042            /**
043             * Return the StickyWebResponse data as raw text.
044             * @return
045             */
046            public String asText() {
047                    return text;
048            }
049            
050            /**
051             * Return the data as structured JSON data, encoded in a Map.
052             * @return
053             * @throws ParseException
054             * @throws StickyWebJsonResponseParseException 
055             */
056            public Map<String, Object> asJSON() throws StickyWebJsonResponseParseException {
057                    return JsonConverter.convertToMap(text);
058            }
059            
060            /**
061             * Return the data as structured JSON data, encoded in a List.
062             * @return
063             * @throws ParseException
064             */
065            public List<Object> asJSONArray() throws StickyWebJsonResponseParseException {
066                    return JsonConverter.convertToList(text);
067            }
068            
069            /**
070             * Return the data as CSV, which will be a list of arrays of strings.
071             * @return
072             * @throws IOException
073             */
074            public List<String[]> asCSV() throws IOException {
075                    CSVReader reader = new CSVReader(new StringReader(text));
076                    List<String[]> result = reader.readAll();
077                    reader.close();
078                    return result;
079            }
080            
081            /**
082             * Return the data as CSV, which will be a list of arrays of strings. Features options for parsing the CSV.
083             * @param separator
084             * @param quotechar
085             * @param escape
086             * @return
087             * @throws IOException
088             */
089            public List<String[]> asCSV(char separator, char quotechar, char escape) throws IOException {
090                    CSVReader reader = new CSVReader(new StringReader(text), separator, quotechar, escape);
091                    List<String[]> result = reader.readAll();
092                    reader.close();
093                    return result;
094            }
095            
096            
097            /**
098             * Returns the text as an XML document, which can then be queried using 
099             * the native <a href='http://docs.oracle.com/javase/tutorial/jaxp/xslt/xpath.html'>XPath</a>. 
100             * @return
101             * @throws Exception
102             */
103            public Document asXML() throws Exception {
104                    return XMLConverter.convert(text);
105            }
106            
107            /**
108             * Returns the text as an HTML document, which can then be queried using the method evaluateXPath, or any of
109             * the methods related to the <a href='http://htmlcleaner.sourceforge.net/doc/index.html'>htmlcleaner</a> library. 
110             * @return
111             * @throws SAXException
112             * @throws IOException
113             */
114            public TagNode asHTML() throws SAXException, IOException {
115                    return HTMLConverter.convert(text); 
116            }
117    
118    }
