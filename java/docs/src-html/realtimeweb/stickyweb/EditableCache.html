001    package realtimeweb.stickyweb;
002    
003    import java.io.OutputStream;
004    import java.util.ArrayList;
005    import java.util.HashMap;
006    import java.util.List;
007    import java.util.Map.Entry;
008    
009    import org.json.simple.JSONValue;
010    
011    import realtimeweb.stickyweb.exceptions.StickyWebDataSourceNotFoundException;
012    import realtimeweb.stickyweb.exceptions.StickyWebDataSourceParseException;
013    import realtimeweb.stickyweb.exceptions.StickyWebLoadDataSourceException;
014    
015    public class EditableCache extends LocalCache {
016    
017            public void setPattern(String key, Pattern pattern) {
018                    if (this.data.containsKey(key)) {
019                            Output element = this.data.get(key);
020                            element.setPattern(pattern);
021                    } else {
022                            this.data.put(key, new Output(pattern));
023                    }
024            }
025    
026            /**
027             * Set the data for this key. The *data* should not start with the pattern.
028             * 
029             * @param key
030             * @param pattern
031             * @param data
032             *            The list of strings that will be returned for this key. Should
033             *            not start with the pattern.
034             */
035            public void setData(String key, Pattern pattern, ArrayList<String> data) {
036                    if (this.data.containsKey(key)) {
037                            Output element = this.data.get(key);
038                            element.setPattern(pattern);
039                    } else {
040                            this.data.put(key, new Output(pattern, data));
041                    }
042            }
043    
044            public void removeData(String key, int index) {
045                    Output element = this.data.get(key);
046                    element.removeData(index);
047            }
048    
049            public void removeAllData(String key) {
050                    Output element = this.data.get(key);
051                    element.removeAllData();
052            }
053    
054            public void addData(String key, String data) {
055                    Output element = this.data.get(key);
056                    element.removeAllData();
057            }
058    
059            public String getData(String key, int index) {
060                    Output element = this.data.get(key);
061                    return element.get(index);
062            }
063    
064            public List<String> getAllData(String key) {
065                    Output element = this.data.get(key);
066                    return element.getAll();
067            }
068            
069            public String saveToStream(OutputStream stream)
070                            throws StickyWebDataSourceNotFoundException,
071                            StickyWebDataSourceParseException, StickyWebLoadDataSourceException {
072                    if (stream == null) {
073                            throw new StickyWebDataSourceNotFoundException(
074                                            "The given OutputStream was null; check to make sure that the stream exists.");
075                    }
076                    HashMap<String, Object> dataSource = new HashMap<String, Object>();
077                    dataSource.put("metadata", this.metadata);
078                    HashMap<String, Object> actualData = new HashMap<String, Object>();
079                    for (Entry<String, Output> entry : this.data.entrySet()) {
080                            actualData.put(entry.getKey(), entry.getValue().recombine());
081                    }
082                    dataSource.put("data", actualData);
083                    return JSONValue.toJSONString(dataSource);
084            }
085    }
