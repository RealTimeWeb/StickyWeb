001    package realtimeweb.stickyweb;
002    
003    import java.util.ArrayList;
004    
005    class Output {
006            /**
007             * Create a new Output object from a list of outputs.
008             * @param value
009             */
010            Output(ArrayList<String> value) {
011                    clock = 0;
012                    data = value;
013            }
014            private ArrayList<String> data;
015            private int clock;
016            
017            /**
018             * Returns the current data, based on the clock. If there is no more data,
019             * then the result is returned based on the pattern:<ul>
020             *      <li>RESTART: turn the clock back to 0 and return first result.</li>
021             *  <li>REPEAT: turn the clock back to the final element and return it again.</li>
022             *  <li>EMPTY: return null, which should be turned into a relevant "empty" element.</li>
023             *  </ul>
024             * @param pattern
025             * @return
026             */
027            String get(Pattern pattern) {
028                    if (!hasMore()) {
029                            switch (pattern) {
030                            case EMPTY:
031                                    return null;
032                            case RESTART:
033                                    clock = 0;
034                                    break;
035                            default: case REPEAT:
036                                    clock = data.size()-1;
037                                    break;
038                            }
039                    }
040                    return data.get(clock++);
041            }
042            
043            /**
044             * Returns whether or not there are more items for this output.
045             * @return
046             */
047            boolean hasMore() {
048                    return clock < data.size();
049            }
050            
051            /**
052             * Resets the clock on this input, so that it returns results from the beginning.
053             */
054            void reset() {
055                    clock = 0;
056            }
057    }
