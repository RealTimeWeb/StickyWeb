001    package realtimeweb.stickyweb;
002    
003    import java.io.IOException;
004    import java.io.InputStream;
005    import java.net.URISyntaxException;
006    import java.util.Map;
007    
008    import realtimeweb.stickyweb.exceptions.StickyWebDataSourceNotFoundException;
009    import realtimeweb.stickyweb.exceptions.StickyWebDataSourceParseException;
010    import realtimeweb.stickyweb.exceptions.StickyWebLoadDataSourceException;
011    
012    public class StickyWeb {
013            
014            private LocalCache cache;
015            
016            public StickyWeb() {
017                    cache = new LocalCache();
018            }
019            
020            public StickyWeb(InputStream dataSource) throws StickyWebDataSourceNotFoundException, StickyWebDataSourceParseException, StickyWebLoadDataSourceException {
021                    if (dataSource != null) {
022                            cache = new LocalCache(dataSource);
023                    } else {
024                            cache = null;
025                    }
026            }
027            
028            public StickyWebRequest get(String url, Map<String, String> arguments) throws IllegalStateException, IOException, URISyntaxException {
029                    return new StickyWebRequest(url, arguments, cache, Protocol.GET);
030            }
031            
032            public StickyWebRequest post(String url, Map<String, String> arguments) throws IllegalStateException, IOException, URISyntaxException {
033                    return new StickyWebRequest(url, arguments, cache, Protocol.POST);
034            }
035            
036            public StickyWebRequest delete(String url, Map<String, String> arguments) throws IllegalStateException, IOException, URISyntaxException {
037                    return new StickyWebRequest(url, arguments, cache, Protocol.DELETE);
038            }
039            
040            public StickyWebRequest put(String url, Map<String, String> arguments) throws IllegalStateException, IOException, URISyntaxException {
041                    return new StickyWebRequest(url, arguments, cache, Protocol.PUT);
042            }
043    
044    }
