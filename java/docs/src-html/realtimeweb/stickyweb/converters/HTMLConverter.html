001    package realtimeweb.stickyweb.converters;
002    
003    import java.io.IOException;
004    
005    import org.htmlcleaner.HtmlCleaner;
006    import org.htmlcleaner.TagNode;
007    import org.xml.sax.SAXException;
008    
009    public class HTMLConverter {
010            
011            private static HtmlCleaner cleaner = new HtmlCleaner();
012            
013            public static TagNode convert(String data) throws SAXException, IOException {
014                    return cleaner.clean(data);
015            }
016    }
