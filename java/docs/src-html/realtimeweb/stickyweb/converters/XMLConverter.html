001    package realtimeweb.stickyweb.converters;
002    
003    import java.io.IOException;
004    import java.io.StringReader;
005    
006    import org.w3c.dom.Document;
007    import org.xml.sax.InputSource;
008    import org.xml.sax.SAXException;
009    
010    import com.sun.org.apache.xerces.internal.parsers.DOMParser;
011    
012    public class XMLConverter {
013            
014            private static DOMParser parser = new DOMParser();
015    
016            public static Document convert(String data) throws SAXException, IOException {
017                    parser.parse(new InputSource(new StringReader(data)));
018                    Document doc = parser.getDocument();
019                    return doc;
020            }
021    
022    }
